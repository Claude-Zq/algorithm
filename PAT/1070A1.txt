核心：长绳子对折对总长度的影响大于短绳子，按升序排列后处理即可

```c++
#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
	int n;
	vector<int> v;
	scanf("%d", &n);
	v.resize(n);
	for (int i = 0; i < n; i++) scanf("%d", &v[i]);
	sort(v.begin(), v.end());/*排序*/
	int len = v[0];/*len是int还是doule都行*/
	for (int i = 1; i < n; i++) len = (len + v[i]) / 2;
    printf("%d", len);
}
```
