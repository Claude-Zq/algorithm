## 插入排序

```c++
void InsertSort(int a[], int n) {

	for (int i = 1; i < n; i++) {
		int temp = a[i], j = i;//可以把j理解为空位的索引
		while (j >= 1 && temp < a[j-1]) { a[j] = a[j - 1]; j--; }
		a[j] = temp;
	}
}
```





## 区间贪心



### 先选左端点最大

```c++
#include<cstdio>
#include<algorithm>
using namespace std;

const int maxn = 100;

struct Interval{
	int x, y;
};

Interval I[maxn];

bool cmp(const Interval& a, const Interval& b) {
	if (a.x != b.x) return a.x > b.x;
	return a.y < b.y;
}

int main()
{
	int n;
	scanf("%d", &n);
	for (int i = 0; i < n; i++) scanf("%d %d", &I[i].x, &I[i].y);

	sort(I, I + n, cmp);/*排序*/

	int ans = 1, lastX = I[0].x;
	for (int i = 1; i < n; i++) if (I[i].y <= lastX) {ans++;lastX = I[i].x;}
    
	printf("%d\n", ans);
	return 0;
}

```



### 先选右端点最小

```c++
#include<cstdio>
#include<algorithm>
using namespace std;

const int maxn = 100;
struct Interval{int x, y;};

Interval I[maxn];

bool cmp(const Interval& a, const Interval& b) {
	if (a.y != b.y) return a.y < b.y;
	return a.x > b.x;
}

int main()
{

	int n;
	scanf("%d", &n);
	for (int i = 0; i < n; i++) scanf("%d %d", &I[i].x, &I[i].y);

	sort(I, I + n, cmp);

	int ans = 1, lastY = I[0].y;
	for (int i = 1; i < n; i++) {
		if (I[i].x >= lastY) {
			ans++;
			lastY = I[i].y;
		}
	}
	
	printf("%d\n", ans);
	
	return 0;
}

```
